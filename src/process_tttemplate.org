#+BEGIN_SRC
export PROCESS_TTTEMPLATE_PID=`echo -n $$`
if [ "$_PHP001" == "" ]
then
    export _PHP001=`which php`;
    echo
fi
if [ "$_PHP001" == "" ]
then
 echo "Fatal error,no php"
 #return 1
 #exit 1
 kill -9 $PROCESS_TTTEMPLATE_PID
fi

echo "nr args:$#"
if [ "$#" == "3" ]
then
 echo "arg 1:$1"
 echo "arg 2:$2"
 echo "arg 2:$3"
else
 echo "$0:expected 3 arguments"
 #return 1
 #exit 1
 kill -9 $PROCESS_TTTEMPLATE_PID
fi

bashinclude01="$1"
phpinclude01="$2"
destdir01="$3"

if test -e "$destdir01"
then
 echo "Fatal error,file exists:$destdir01"
 #return 1
 #exit 1
 kill -9 $PROCESS_TTTEMPLATE_PID
fi

mkdir "$destdir01"

cp -rf * "$destdir01"/.

cd "$destdir01"
